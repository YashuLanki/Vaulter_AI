# Vaulter AI Property Intelligence System
# Install all dependencies with: pip install -r requirements.txt
#
# Recommended Python: 3.11 or 3.12. A much newer interpreter may not yet
# have ready-to-use installs for packages like chromadb/sentence-transformers
# on every platform, forcing a slow or broken from-source build instead --
# setup_wizard.py checks this and warns if it applies to you.

# ── Stage 1: PDF Ingestion ─────────────────────────────────────────
pdfplumber          # PDF text and table extraction
pytesseract         # Python wrapper for Tesseract OCR
pdf2image           # Convert PDF pages to images for OCR
chromadb>=0.5.20        # Vector database — 0.5.20+ fixes RustBindingsAPI concurrent access bug
sentence-transformers   # Real semantic embeddings (all-MiniLM-L6-v2) for RAG search quality
numpy               # Numerical computing (used for embeddings)
watchdog            # File system monitoring for watched_folder
filelock            # Cross-process file locking for shared registry/cache files
openpyxl            # Excel .xlsx file reading (Stage 1 + email attachments)
xlrd                # Excel .xls file reading (older format)
pandas              # CSV file reading and data handling
rapidfuzz           # Fuzzy string matching for property name detection

# ── Stage 2: Web & Email Pipeline ─────────────────────────────────
beautifulsoup4      # Web scraping
requests            # HTTP requests for web scraping and Graph API
msal                # Microsoft Authentication Library (Outlook OAuth)
apscheduler         # Background scheduler for automated jobs
mammoth             # Extract text from Word (.docx) email attachments
python-pptx         # Extract text from PowerPoint (.pptx) email attachments
python-dotenv       # Load credentials from .env file
Pillow              # Image handling for OCR on image email attachments

# ── Stage 3: Claude Analysis + MCP Server ─────────────────────────
anthropic           # Anthropic Claude API
mcp[cli]            # MCP server framework (local stdio, launched by Claude Desktop)
uvicorn             # ASGI server (required by MCP for SSE transport)

# ── Stage 4: Speech-to-Knowledge ──────────────────────────────────
openai              # Whisper API for audio transcription

# ── CoStar Listing Screener (analysis/screening/) ─────────────────
# No new packages needed — pandas, openpyxl, python-dotenv, anthropic,
# and requests above already cover everything the 4-phase screening
# pipeline and Google Maps Platform verification calls require.
